281  {
282  struct ::addrinfo hints;
283  std::memset(&hints, 0, sizeof(hints));
284  hints.ai_family = AF_UNSPEC;
285  hints.ai_socktype = SOCK_STREAM;
286  hints.ai_protocol = 0; // ANY
287 
288  // additional flags
289  hints.ai_flags |= ai_hint_flags;
290 
291  try
292  {
293  const _internal::gai_guard gai(host, serv, hints);
294 
295  for (struct ::addrinfo* it = gai.result; it != null; it = it->ai_next)
296  {
297  ident_t socket = ::socket(it->ai_family, it->ai_socktype, it->ai_protocol);
298  if (socket < 0)
299  {
300  throw syscall_failed();
301  }
302 
303  if ((*func)(socket, it->ai_addr, it->ai_addrlen) < 0)
304  {

311 
312  throw syscall_failed();
