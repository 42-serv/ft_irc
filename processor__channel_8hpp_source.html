41 
42  std::size_t i = 0;
43  foreach (ft::irc::message::param_vector::const_iterator, it, channel_names)
44  {
45  const std::string& channel_name = *it;
46 
47  std::string key;
48  if (i < keys.size())
49  {
50  key = keys[i];
51  }
52  i++;
53 
54  if (user.is_channel_member(channel_name))
55  {
56  continue;
57  }
58 
59  if (!ft::irc::string_utils::is_valid_channelname(channel_name))
60  {

63  }
64

66  {

68  continue;
69  }
70

244  void update_user_mode(ft::irc::user& user, const ft::shared_ptr<ft::irc::user>& target, const std::string& mode_str) const
245  {
246  enum
247  {
248  NONE,
249  ADD,
250  REMOVE
251  } state = NONE;
252  foreach (std::string::const_iterator, it, mode_str)
253  {
254  std::string::value_type c = *it;
255  if (c == '+')
256  {
257  state = ADD;
258  }
259  else if (c == '-')
260  {
261  state = REMOVE;
262  }
263  else
264  {
265  if (state == NONE)
266  {
267  // nothing
268  continue;
269  }
270 
271  bool add = state == ADD;
272  switch (c)
273  {
274  case 'o':
275  if (add)
276  {
277  // ignore self opping
278  continue;
279  }
280  target->store_mode(ft::irc::user::USER_MODE_OPERATOR, add);
281  break;
282  case 'i':
283  target->store_mode(ft::irc::user::USER_MODE_INVISIBLE, add);
284  break;
285  case 'a':
286  target->store_mode(ft::irc::user::USER_MODE_AWAY, add);
287  break;
288  default:
289  user.send_message(ft::irc::make_error::user_mode_unknown_flag());
290  }

295  bool update_channel_mode(ft::irc::user& user, const ft::shared_ptr<ft::irc::channel>& channel, const ft::irc::message& message, bool can_modify) const
296  {
297  const std::string& mode_str = message[1];

299  {
300  NONE,
301  ADD,
302  REMOVE
303  } state = NONE;
304  std::size_t param_index = this->get_min_params() + 1;
305  foreach (std::string::const_iterator, it, mode_str)
306  {
307  std::string::value_type c = *it;
308  if (c == '+')
309  {
310  state = ADD;
311  }
312  else if (c == '-')
313  {
314  state = REMOVE;
315  }
316  else
317  {
318  if (state == NONE)
319  {
320  switch (c)
321  {
322  case 'b':
323  // Query ban list
324  channel->send_ban_list(user);
325  break;
326  }
327  continue;
328  }
329  else if (!can_modify)
330  {
331  return false;
332  }
333 
334  bool add = state == ADD;
335  switch (c)
336  {
337  case 'o':
338  if (message.param_size() <= param_index)
339  {
340  break;
341  }
342

344  const std::string& nick = message[param_index++];
345  channel->modify_member_mode(nick, ft::irc::channel::member::MEMBER_MODE_OPERATOR, add);
346  }
347  break;
348  case 'p':
349  channel->store_mode(ft::irc::channel::CHANNEL_MODE_PRIVATE, add);
350  break;
351  case 's':
352  channel->store_mode(ft::irc::channel::CHANNEL_MODE_SECRET, add);
353  break;
354  case 'i':
355  channel->store_mode(ft::irc::channel::CHANNEL_MODE_INVITE_ONLY, add);
356  break;
357  case 't':
358  channel->store_mode(ft::irc::channel::CHANNEL_MODE_TOPIC_LIMIT, add);
359  break;
360  case 'n':
361  channel->store_mode(ft::irc::channel::CHANNEL_MODE_NO_PRIVMSG, add);
362  break;
363  case 'm':
364  channel->store_mode(ft::irc::channel::CHANNEL_MODE_MODERATED, add);
365  break;
366  case 'l':
367  if (add)
368  {
369  if (message.param_size() <= param_index)
370  {
371  break;
372  }
373 
374  const std::string& arg = message[param_index++];
375  std::istringstream iss(arg);
376 
377  std::size_t limit;
378  iss >> limit;
379  if (!iss)
380  {
381  limit = 0;
382  }
383 
384  channel->store_limit(limit);
385  }
386  else
387  {
388  channel->store_limit(0);
389  }
390 
391  channel->store_mode(ft::irc::channel::CHANNEL_MODE_LIMIT, add);
392  break;
393  case 'b':
394  {
395  if (message.param_size() <= param_index)
396  {
397  break;
398  }
399 
400  const std::string& mask = message[param_index++];
401  if (add)
402  {
403  channel->ban(mask);
404  }
405  else
406  {
407  channel->unban(mask);
408  }
409  break;
410  }
411  case 'v':
412  if (message.param_size() <= param_index)
413  {
414  break;
415  }
416 
417  {
418  const std::string& nick = message[param_index++];
419  channel->modify_member_mode(nick, ft::irc::channel::member::MEMBER_MODE_VOICE, add);
420  }
421  break;
422  case 'k':
423  if (add)
424  {
425  if (message.param_size() <= param_index)
426  {
427  break;
428  }
429 
430  const std::string& key = message[param_index++];
431 
432  if (!key.empty())
433  {
434  channel->store_key(key);
435  }
436  }
437  else

439  channel->store_key("");
440  }

442  channel->store_mode(ft::irc::channel::CHANNEL_MODE_KEY, add);
