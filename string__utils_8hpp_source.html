222  static inline bool match_mask(const std::string& pattern_str, const std::string& name_str)
223  {
224  std::string::const_iterator pattern = pattern_str.begin();
225  std::string::const_iterator name = name_str.begin();
226  std::string::const_iterator pattern_backup = pattern_str.end();
227  std::string::const_iterator name_backup = name_str.end();
228 
229  while (pattern != pattern_str.end() || name != name_str.end())
230  {
231  // continue when match
232  if (pattern != pattern_str.end())
233  {
234  if (*pattern == '*')
235  {
236  // *
237  pattern_backup = pattern;
238  name_backup = name;
239  pattern++;
240  continue;
241  }
242  else if (*pattern == '?')
243  {
244  // ?
245  if (name != name_str.end())
246  {
247  pattern++;
248  name++;
249  continue;
250  }
251  }
252  else
253  {
254  // character
255  if (name != name_str.end() && *pattern == *name)
256  {
257  pattern++;
258  name++;
259  continue;
260  }
261  }
262  }
263 
264  // mismatch
265  if (name_backup != name_str.end())
266  {
267  // restart from last *
268  pattern = pattern_backup;
269  name = name_backup;
270  name++;
271  continue;
272  }
273 
274  return false;
275  }
276 
277  return true;
278  }
